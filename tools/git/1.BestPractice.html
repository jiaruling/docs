# 实战

# 1. ubuntu 安装 nodejs && npm

# 更新包管理器
$ sudo apt update

# 安装指定版本的nodejs(nodejs 12.x 版本)
$ cd ~
$ curl -sL https://deb.nodesource.com/setup_12.x | sudo bash -
$ sudo apt install nodejs

# 安装新版本的nodejs
$ cd ~
$ curl -sL https://deb.nodesource.com/setup | sudo bash -
$ sudo apt install nodejs

# 安装 yarn 包管理器
$ curl -sS https://dl.yarnpkg.com/debian/pubkey.gpg | sudo apt-key add -
$ echo "deb https://dl.yarnpkg.com/debian/ stable main" | sudo tee /etc/apt/sources.list.d/yarn.lis
$ sudo apt install yarn

# 查看是否安装成功
$ node --version
$ npm --version
$ yarn --version

# 卸载nodejs
$ sudo apt remove --purge nodejs # 卸载nodejs并清除配置文件
$ sudo apt autoremove # 移除和 nodejs 一起安装但是现在没有被使用的包

# 2. 安装 Commitizen 并使用

在项目中使用（不推荐)

在项目中会产生与项目无关的文件【node_modules package-lock.json package.json】

# 1.通过 npm 安装 cz 工具
$ sudo npm install -g commitizen

# 2.创建代码仓库
$ mkdir git-demo
$ cd git-demo
$ ls

# 3.安装适配器
$ commitizen init cz-conventional-changelog --save --save-exact
$ ls
node_modules  package-lock.json  package.json
$ cat package.json
{
  "devDependencies": {
    "cz-conventional-changelog": "^3.3.0"
  },
  "config": {
    "commitizen": {
      "path": "./node_modules/cz-conventional-changelog"
    }
  }
}

# 4.初始化仓库并添加文件
$ git init
$ vim test.py  # 文件中写入 print("hello world")
$ git add test.py
$ git status

全局使用（推荐）

不会产生和项目无关的文件

# 1.通过 npm 安装 cz 工具
$ sudo npm install -g commitizen

# 2.切换到~目录
$ cd ~ 
# 3.在~目录安装适配器
$ commitizen init cz-conventional-changelog --save --save-exact
$ ls
git-demo  node_modules  package-lock.json  package.json
# 4.编辑package.json文件
$ vim package.json # 打开 package.json, 去除 ./node_modules/
$ cat package.json
{
  "devDependencies": {
    "cz-conventional-changelog": "^3.3.0"
  },
  "config": {
    "commitizen": {
      "path": "cz-conventional-changelog"
    }
  }
}
# 5.创建代码仓库并完成初始化
$ mkdir git-demo2
$ cd git-demo2/
$ git init
$ vim test2.py # 文件中写入 print("hello demo2")
$ git add test2.py
$ git status
